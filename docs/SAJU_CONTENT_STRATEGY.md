  ): StrengthInterpretation {
    const level = this.getStrengthLevel(score);
    const template = strengthTemplates[level];
    
    // reasonsë¥¼ ìì—°ì–´ë¡œ ë³€í™˜
    const reasonText = reasons.map(r => this.formatReason(r)).join('\n');
    
    return {
      ...template,
      score,

  }

  static interpretElements(
    elements: Record<string, number>
  ): ElementInterpretation[] {
    const total = Object.values(elements).reduce((a, b) => a + b, 0);
    
    return Object.entries(elements).map(([element, count]) => {
      const ratio = (count / total) * 100;
      const status = this.getElementStatus(ratio);
      const base = elementDatabase[element];
      
      return {
        element,
        count,
        ratio,
        status,
        ...base,
        description: this.generateElementDescription(base, status)
      };
    });
  }
  
  // 4. ì‹­ì‹  í•´ì„ ìƒì„±
  static interpretTenGods(
    tenGods: TenGodDistribution
  ): TenGodInterpretation[] {
    return Object.entries(tenGods)
      .sort(([,a], [,b]) => b - a) // ë§ì€ ìˆœìœ¼ë¡œ ì •ë ¬
      .slice(0, 3) // ìƒìœ„ 3ê°œë§Œ
      .map(([tenGod, count]) => ({
        tenGod,
        count,
        position: count >= 2 ? 'strong' : count === 1 ? 'moderate' : 'weak',
        ...tenGodDatabase[tenGod]
      }));
  }
  
  // 5. ëŒ€ìš´ í•´ì„ ìƒì„±

    daeun: DaeunItem,
    dayMaster: string
  ): DaeunInterpretation {

    const tenGod = calculateTenGod(dayMaster, daeun.stem);
    
    return {
      age: daeun.age,
      ganji: daeun.ganji,
      element: daeun.element,
      tenGod,
      ...base,
      strategy: this.generateDaeunStrategy(tenGod, daeun.element)
    };
  }
